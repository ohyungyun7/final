CREATE TABLE MEMBER1(
    MEM_NO NUMBER(8) NOT NULL,
    MEM_ID VARCHAR2(100) NOT NULL,
    MEM_PW VARCHAR2(100) NOT NULL,
    MEM_NAME VARCHAR2(100) NOT NULL,
    MEM_BIRTH VARCHAR2(100) NOT NULL,
    MEM_GENDER VARCHAR2(30) NOT NULL,
    MEM_FAVORITE_SUBJECT VARCHAR2(100) NOT NULL,
    MEM_GRADE NUMBER(8) DEFAULT 1 NULL,
    CONSTRAINT pk_member1 PRIMARY KEY(MEM_NO)
);

CREATE TABLE NOTICE1(
    BO_NO NUMBER(8) NOT NULL,
    BO_TITLE VARCHAR2(200) NOT NULL,
    BO_CONTENT VARCHAR2(2000) NOT NULL,
    BO_WRITER VARCHAR2(100) NOT NULL,
    BO_HIT NUMBER(8) DEFAULT 0 NULL,
    BO_DATE DATE DEFAULT SYSDATE NULL,
    CONSTRAINT pk_notice1 PRIMARY KEY(BO_NO)
);

CREATE TABLE FREE1(
    BO_NO NUMBER(8) NOT NULL,
    BO_TITLE VARCHAR2(200) NOT NULL,
    BO_CONTENT VARCHAR2(2000) NOT NULL,
    BO_WRITER VARCHAR2(100) NOT NULL,
    BO_HIT NUMBER(8) DEFAULT 0 NULL,
    BO_DATE DATE DEFAULT SYSDATE NULL,
    CONSTRAINT pk_free1 PRIMARY KEY(BO_NO)
);

CREATE TABLE INQUIRY1(
    BO_NO NUMBER(8) NOT NULL,
    BO_TITLE VARCHAR2(200) NOT NULL,
    BO_CONTENT VARCHAR2(2000) NOT NULL,
    BO_WRITER VARCHAR2(100) NOT NULL,
    BO_HIT NUMBER(8) DEFAULT 0 NULL,
    BO_DATE DATE DEFAULT SYSDATE NULL,
    CONSTRAINT pk_inquiry1 PRIMARY KEY(BO_NO)
);

CREATE TABLE EDUINFORMAION1(
    EDU_NO NUMBER(8) NOT NULL,
    MEM_NO NUMBER(8) NOT NULL,
    EDU_BASICJAVA CHAR(2) DEFAULT 'X' NULL,
    EDU_ORACLE CHAR(2) DEFAULT 'X' NULL,
    EDU_MASTERJAVA CHAR(2) DEFAULT 'X' NULL,
    EDU_HTML CHAR(2) DEFAULT 'X' NULL,
    EDU_CSS CHAR(2) DEFAULT 'X' NULL,
    EDU_JAVASCRIPT CHAR(2) DEFAULT 'X' NULL,
    EDU_JQUERY CHAR(2) DEFAULT 'X' NULL,
    EDU_JSP CHAR(2) DEFAULT 'X' NULL,
    EDU_SERVLET CHAR(2) DEFAULT 'X' NULL,
    EDU_SPRING CHAR(2) DEFAULT 'X' NULL,
    CONSTRAINT pk_eduinformation PRIMARY KEY(EDU_NO),
    CONSTRAINT fk_eduinformaion_member FOREIGN KEY(MEM_NO) REFERENCES MEMBER1(MEM_NO)
);

CREATE SEQUENCE SEQ_FREE1 INCREMENT BY 1 START WITH 1;
